# Initialize a Next.js Project with pnpm

# Install Packages

pnpm create next-app@latest <project_name> --ts --eslint --tailwind --app --src-dir --use-pnpm --import-alias "@/*" --yes
npx skills add https://github.com/vercel-labs/next-skills --skill next-best-practices



project_name: The name of your Next.js project. Replace `<project_name>` with your desired project name.
